# Collection: users + userId is uid
users/{userId} {
  name: string,
  email: string,
  phone: string,
  image_url: string,
  saved_at: timestamp
}

      # Google_tokens:
      users/{userId}/google_tokens/{tokenId} {
        access_token: string,
        refresh_token: string,
        token_expiry: timestamp,
        scope: string,
        saved_at: timestamp
      }

# Business:
businesses/{businessId} {
  user_id: string,
  category_ids: string[],
  name: string,
  description: string,
  location: string,
  contact_email: string,
  contact_phone: string,
  google_calendar_id: string,
  image_url: string,
  saved_at: timestamp
}

      # slots
      businesses/{businessId}/slots/{slotid} {
        name: string,
        description: string,
        saved_at: timestamp,
        duration_min: number,
        category_id: string,
        image_url: string,
        is_active: boolean
      }

# Categories: 
categories/{category_id} {
  name: string,
  description: string,
}


# Appointments:
appointments/{appointmentId} {
  user_id: string,
  slot_id: string,
  business_id: string
  start_time: timestamp,
  saved_at: timestamp,
  google_event_id: string,
  status: string,  // enum: 'pending' | 'confirmed' | 'cancelled' | 'completed'
  note: string
}
